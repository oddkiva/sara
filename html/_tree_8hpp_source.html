426  void remove_child(Node *child)
427  {
428  Node *prev_sibling = child->_prev_sibling;
429  Node *next_sibling = child->_next_sibling;
430 
431  // Modify myself, the parent of this child.
432  if (_first_child == child)
433  _first_child = next_sibling;
434  if (_last_child == child)
435  _last_child = prev_sibling;
436 
437  // Modify the siblings.
438  prev_sibling->_next_sibling = next_sibling;
439  next_sibling->_prev_sibling = prev_sibling;

441  // Make this child an orphan.

444  child->_prev_sibling = 0;
