196  const self_type& operator=(const self_type& M)
197  {
198  // We must be careful when (M == *this)!
199  if (!is_wrapped_data_ && begin_ != M.data())
200  {
201  delete[] begin_;
202  begin_ = 0;
203  end_ = 0;
204  }
205 
206  // Allocate new data.
207  T *newData = new T[M.size()];
208  sizes_ = M.sizes();
209  strides_ = M.strides();
210  is_wrapped_data_ = false;
211 
212  if (newData)
213  std::copy(M.data(), M.data() + M.size(), newData);
214 
215  // We must be careful when (M == *this)!
216  if (begin_ == M.data())
217  delete[] begin_;
218  begin_ = newData;
219  end_ = begin_ + M.size();
220 
221  return *this;
222  }
