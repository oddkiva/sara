391  {
392  return (static_cast<double>(value) - ChannelTraits<T>::doubleMin())
393  / ChannelTraits<T>::doubleRange();
394  }
396  template <typename T, int N>
397  inline Matrix<double, N, 1> getRescaledColor64f(const Matrix<T, N, 1>& color)
398  {
399  Matrix<double, N, 1> min; min.fill(ChannelTraits<T>::doubleMin());
400  Matrix<double, N, 1> max; max.fill(ChannelTraits<T>::doubleMax());
401  return (color.template cast<double>() - min).cwiseQuotient(max - min);
402  }

433  template <typename T>
434  inline double rgb2gray64f(const Matrix<T, 3, 1>& rgb)
435  {
436  Matrix<double, 3, 1> rgb64f(getRescaledColor64f(rgb));
437  return 0.3*rgb64f[0] + 0.59*rgb64f[1] + 0.11*rgb64f[2];
438  }
440  template <typename T>
441  inline Matrix<double, 3, 1> gray2rgb64f(T gray)
442  {
443  double gray64f = getRescaledChannel64f(gray);
444  Matrix<double, 3, 1> rgb64f;
445  rgb64f.fill(gray64f);
446  return rgb64f;
447  }
449  template<typename T>
450  inline Vector3d rgb2yuv64f(const Matrix<T, 3, 1>& rgb)
451  {
452  Matrix<double, 3, 1> rgb64f(getRescaledColor64f(rgb));
453 
454  double y = .299*rgb64f[0] + .587*rgb64f[1] + .114*rgb64f[2];
455  return Vector3d(y, .492*(rgb64f[2]-y), .877*(rgb64f[0]-y));
456  }
458  template<typename T>
459  inline Vector3d yuv2rgb64f(const Matrix<T, 3, 1>& yuv)
460  {
461  Vector3d yuv64f(getRescaledColor64f(yuv));
462  double r = yuv64f[2]/.877 + yuv64f[0];
463  double b = yuv64f[1]/.492 + yuv64f[0];
464  double g = (yuv64f[0] - .299*r - .114*b) / .587;
465  return Matrix<double, 3, 1>(r, g, b);
466  }
